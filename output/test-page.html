        public override ClrObject Object
        {
            get
            {
                if (_object.Address != 0)
                    return _object;

                ClrObject obj = base.Object;
                if (obj.Type is not null)
                {
                    _object = obj;
                }
                else
                {
                    ClrObject prev = _heap.FindPreviousObjectOnSegment(obj);
                    if (prev.IsValid && prev <= obj && obj < prev + prev.Size)
                    {
                        _object = prev;
                    }
                    else
                    {
                        _object = obj;
                    }
                }

                return _object;
            }
        }
